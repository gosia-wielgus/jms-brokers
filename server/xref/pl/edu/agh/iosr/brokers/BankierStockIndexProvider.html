46  			// OK
47  		}

50  	private void checkForUpdates() {
51  		try {
52  			System.getProperties().put( "proxySet", "true" );
53  			System.getProperties().put( "proxyHost", "wwwgate0.mot.com");
54  			System.getProperties().put( "proxyPort", "1080");
55  			
56  			doc = Jsoup.connect(BANKIER_URL).get();
57  			String updateKey = "Aktualizacja:";
58  			Date date = longFormatter.parse(doc.getElementsContainingOwnText(updateKey).text().replace(updateKey, "").trim());
59  			
60  			//if(date.after(lastDate)){
61  				//System.out.println(date);
62  				lastDate = date;
63  				dispatchAll();
64  			//}
65  		} catch (IOException e) {
66  			logger.error("Couldnt connect with: " + BANKIER_URL);
67  		} catch (ParseException e) {
68  			logger.error("Unsupported page format: " + e.getMessage());
69  		}
70  	}

72  	private void dispatchAll() throws ParseException {		
73  		for (int i=0; i<indices.size(); i++) {
74  			StockIndex index = indices.get(i);
75  			index = updateStockIndex(index);
76  			logger.info("Dispatched "+index);
77  			dispatch(index);
78  			indices.set(i, index);
79  		}
80  	}

82  	private StockIndex updateStockIndex(StockIndex index) throws ParseException {
83  		Elements rows = doc.select("tr#noto:has(td:eq(0) > a[title=" + index.getName() + "])");
84  		if (rows.size() != 1)
85  			throw new ParseException("Could not find index " + index.getName(), 0);
86  		Element row = rows.get(0);
87  		boolean loss = "quoteDown".equals(row.child(1).attr("class"));
88  		BigDecimal value = new BigDecimal(row.child(1).text());
89  		BigDecimal change = new BigDecimal(row.child(2).text());
90  		if (loss)
91  			change = change.multiply(new BigDecimal("-1"));
92  		Date date = shortFormatter.parse(row.child(7).text());	
93  		return new StockIndex(index.getKey(), index.getName(), value, change, date.getTime());
94  	}
