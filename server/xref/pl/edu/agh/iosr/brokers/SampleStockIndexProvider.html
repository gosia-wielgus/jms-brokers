20  	Random random = new Random();

22  	@Override
23  	public void run() {
24  		try {
25  			while (!Thread.interrupted()) {
26  				dispatchAll();

30  			// OK
31  		}

34  	private void dispatchAll() {
35  		for (int i=0; i<indices.size(); i++) {
36  			StockIndex index = indices.get(i);
37  			index = updateStockIndex(index);
38  			dispatch(index);
39  			indices.set(i, index);
40  		}
41  	}
42  	
43  	private StockIndex updateStockIndex(StockIndex index) {
44  		BigDecimal change = new BigDecimal(random.nextGaussian());
45  		BigDecimal value = index.getValue().add(change);
46  		value = value.abs();
47  		value = value.round(new MathContext(8));
48  		change = value.subtract(index.getValue()).add(index.getChange());
49  		return new StockIndex(index.getKey(), index.getName(), value, change, new Date().getTime());
50  	}
