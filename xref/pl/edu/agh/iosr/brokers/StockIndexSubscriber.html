27  	public void start() throws JMSException {
28          // Create a ConnectionFactory
29          ActiveMQConnectionFactory connectionFactory = new ActiveMQConnectionFactory(url);
30  
31          // Create a Connection
32          connection = connectionFactory.createConnection();
33          connection.start();
34  
35          // Create a Session
36          Session session = connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
37  
38          String topics = "";
39          for (String name : stockNames) {
40          	topics += ",brokers."+name;
41          }
42          topics = topics.substring(1);
43          // Create the destination (Topic or Queue)
44          Topic destination = session.createTopic(topics);
45  
46          // Create a MessageConsumer from the Session to the Topic or Queue
47          consumer = session.createConsumer(destination);
48  	}
49  	
50  	public void clean() throws JMSException {
51  		consumer.close();
52          session.close();
53          connection.close();
54  	}
55

57  		ObjectMessage message = (ObjectMessage)consumer.receive(1000);
